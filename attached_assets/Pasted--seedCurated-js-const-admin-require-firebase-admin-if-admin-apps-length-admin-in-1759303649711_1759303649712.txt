async function seedCurated() {
  console.log("🌱 Seeding curated demo stories...");

  for (const story of curatedStories) {
    const storyRef = db.collection("stories").doc();

    await storyRef.set({
      ...story,
      authorId: "demo_user",
      authorName: "Demo User",
      authorAvatar: "https://i.pravatar.cc/150?img=5",
      likes: Math.floor(Math.random() * 50),
      saves: Math.floor(Math.random() * 20),
      commentsCount: Math.floor(Math.random() * 10),
      trendingScore: Math.random().toFixed(2),
      createdAt: admin.firestore.FieldValue.serverTimestamp(),
      isSeed: "curated", // ✅ mark as curated for easy cleanup
    });

    console.log(`📝 Created curated story: ${story.title}`);
  }

  console.log("✅ Curated stories seeded successfully!");
  process.exit(0);
}

seedCurated().catch((err) => {
  console.error("❌ Error seeding curated stories:", err);
  process.exit(1);
});
